#define TYPE __int64
#define maxn  1002

int cas , nx , ny , m;

int g[maxn][maxn] , s[maxn];

int G()
{
        int t;
        char c;
        while( ' ' == ( c = getchar() ) || c == '\n' );
        t = c&15;
        while( '0' <= ( c = getchar() ) && c <= '9' )
                t = t*10 + (c&15);
        return t;
}

void solve()
{
        int i , j , x , y;
        TYPE sum = 0;

        scanf( "%d %d %d" , &nx , & ny , &m );

        for( i = 1; i <= ny;i++ )
        {
                for( j = 1;j <= nx;j++ )
                        g[j][i] = 0;
                s[i] = 0;
        }

        while( m-- )
        {
                x = G(); y = G();
                g[x][y] ++;
        }

        for( i = 1;i <= nx;i++ )
        {
                x = 0;
                for( j = 1;j <= ny;j++ )
                {
                        sum += x * s[j];
                        if( g[i][j] )
                        {
                                x += g[i][j];
                                s[j] += g[i][j];
                        }
                }
        }
        printf( "%I64d\n" , sum );
}

int main()
{
        cas;
        scanf( "%d" , & cas );
        for( int i = 1;i <= cas;i++ )
        {
                printf("Test case %d: " , i );
                solve();
        }
    return 0;
